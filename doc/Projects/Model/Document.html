def toParamMAP
        requestBody = Hash.new
        
        if folderId != nil && folderId.to_i > 0
                requestBody["folder_id"] = folderId
        end
        if description != nil
                requestBody["description"] = description
        end
        if tags != nil
                requestBody["tags"] = tags
        end
        if notify != nil && notify.to_i > 0
                requestBody["notify"] = notify
        end
        
        return requestBody
end
