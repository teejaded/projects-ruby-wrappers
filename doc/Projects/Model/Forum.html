    # File projects/model/Forum.rb, line 298
def toParamMAP
        requestBody = Hash.new
        
        if name != nil
                requestBody["name"] = name
        end
        if content != nil
                requestBody["content"] = content
        end
        if categoryId != nil && categoryId.to_i > 0
                requestBody["category_id"] = categoryId
        end
        if notify != nil
                requestBody["notify"] = notify
        end
        
        
        return requestBody
end
